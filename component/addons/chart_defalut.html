<canvas id="myChart"></canvas>
<!-- //[Chart] - 기본형(막대) -->
<script type="text/javascript">
var context = document.getElementById('myChart').getContext('2d');
var myChart = new Chart(context, {
    type : 'bar', // 차트의 형태
    data : { // 차트에 들어갈 데이터
        labels : [
            //x 축
            '영화','드라마','방송','애니','만화','게임','유틸','문서','교육','이미지'
        ],
        datasets : [
            { //데이터
                label : '회원1', //차트 제목
                fill : false, // line 형태일 때, 선 안쪽을 채우는지 안채우는지
                data : [
                    100,84,92,33,16,68,40,79,55,10 //x축 label에 대응되는 데이터 값
                ],
                backgroundColor : 'rgba(54, 162, 235, 0.2)',
                borderColor : 'rgba(54, 162, 235, 1)',
                borderWidth : 1 //경계선 굵기
            },
            {
                label : '회원2',
                fill : false,
                data : [
                    30,10,50,90,35,90,100,95,20,35,60 //x축 label에 대응되는 데이터 값
                ],
                backgroundColor : 'rgba(255, 0, 0, 0.2)',
                borderColor : 'rgba(255, 0, 0, 1)',
                borderWidth : 1 //경계선 굵기
            }
        ]
    },
    options : {
        scales : {
            xAxes : [ // 가로 타이틀 정보
                {
                    ticks : {
                        beginAtZero : true,
                        fontSize : 14 // 텍스트 크기
                    }
                }
            ],
            yAxes : [ // 세로 타이틀 정보
                {
                    ticks : {
                        beginAtZero : true,
                        min : 0, // 최소값
                        max : 100, // 최대값
                        fontSize : 14 // 텍스트 크기
                    }
                }
            ]
        }
    }
});
</script>
