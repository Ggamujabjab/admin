<!-- [Common] - Jquery Ui -->
<div id="js_datepicker" class="jqueryui_calendar"></div>
<!-- //[Common] - Jquery Ui -->
<script type="text/javascript">
// Datepicker
$("#js_datepicker").datepicker({
    showOtherMonths : true, //빈 공간에 현재월의 앞뒤월의 날짜를 표시
    showMonthAfterYear:true, // 월- 년 순서가아닌 년도 - 월 순서
    changeYear : false, //option값 년 선택 가능
    changeMonth : false, //option값  월 선택 가능
    yearSuffix : " / ", //달력의 년도 부분 뒤 텍스트
    minDate : "-8Y", //최소 선택일자(-1D:하루전, -1M:한달전, -1Y:일년전)
    maxDate : "+8Y" //최대 선택일자(+1D:하루후, -1M:한달후, -1Y:일년후)
});
</script>
