UI COMPONENT SELECT

SELECT

Default (기본 디자인 셀렉트 박스)

Source

닫기 

<!-- [Common] - 셀렉트박스 기본형 -->
<div class="selectType js_selectType">
    <select class="select">
        <option value="https://google.co.kr">영화</option>
        <option value="">드라마</option>
        <option value="">영상</option>
        <option value="">만화</option>
        <option value="">유틸</option>
        <option value="">음악</option>
        <option value="">성인</option>
    </select>
</div>
<!-- //[Common] - 셀렉트박스 기본형 -->
<script type="text/javascript">
// 셀렉트박스 기본형(Select Change 함수실행)
/* ◈◈◈ option ◈◈◈
- effect : "normal" or "fade" ----- 모션 효과
- disable : true or false ----- 셀렉트박스 리스트 비활성화 유무
- alignItem : "top" or "bottom" ----- 활성화 레이어 상하 정렬
- viewCount : "auto" or 숫자 ----- 목록에서 노출될 리스트 갯수(스크롤 생성)
- eventChangeFunc ----- 이벤트 "change" 추가 유무
*/
$(".js_selectType").select({
    eventChangeFunc : function(link){
        console.log(link);
    }
});
</script>

레이어 활성화 모션 적용 및 레이어 목록 활성화 갯수

Source

닫기 

<!-- [Common] - 셀렉트박스 모션 적용 -->
<div id="js_select_motion" class="selectType">
    <select class="select">
        <option value="https://google.co.kr">영화</option>
        <option value="">드라마</option>
        <option value="">영상</option>
        <option value="">만화</option>
        <option value="">유틸</option>
        <option value="">음악</option>
        <option value="">성인</option>
    </select>
</div>
<!-- //[Common] - 셀렉트박스 모션 적용 -->
<script type="text/javascript">
// 셀렉트박스 모션 및 레이어 목록 활성화
$("#js_select_motion").select({ 
    effect : "fade",
    viewCount : 6
});
</script>
