<!-- 레이어 버튼 -->
<a href="#" id="js_button_layer" class="btn_type btn_blue">Layer Button</a>
<!-- //레이어 버튼 -->
<!-- [Common] - Layer Alert -->
<div id="js_layer_alert_script" class="layer_alert_content">
    내용을 넣어주세요<div class=""></div>
    <p class="btn_close"><a href="#">닫기</a></p>
</div>
<!-- //[Common] - Layer Alert -->
<script type="text/javascript">
$("#js_button_layer").on("click", function(e){
    // 기본링크 기능삭제
    e.preventDefault();

    // LayerScratch 레이어
    $.layerScratch({
        defaultTarget : "#js_layer_alert_script",
        opacity : 0.6,
        align : "center",
        autoClose : true,
        nodeAddclass : "scrollHidden"
    });
});
</script>
